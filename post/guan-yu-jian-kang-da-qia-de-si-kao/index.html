#定义规范函数
def _format_addr_(s):
    #拆分用户名和邮箱地址
    name,addr=parseaddr(s)
    #规范化的合并用户名和邮箱地址
    return formataddr((Header(name,'utf-8').encode(),addr))
def send_email():
    msg=MIMEMultipart('alternative')
    #发送内容显示图片的html文件
    if(result == "success"):
        #打卡成功则发送打卡成功图片(1.jpg)
        msg.attach(MIMEText('<html><body><h1>SUCCESS</h1>' +
            '<p><img src="cid:1"></p>' +
            '</body></html>', 'html', 'utf-8'))
        #打开本地文件中的某个图片，并写入将要传入的对象中
        with open(r'1.jpg', 'rb') as f:
            #确认添加的类型为图片格式为jpg，添加头部文件和ID，导入内容，进行编码，链接到邮件
            mine=MIMEBase('image','jpg')
            mine.add_header('Content-ID','1')
            mine.set_payload(f.read())
            encoders.encode_base64(mine)
            msg.attach(mine)
    elif(result == "erro"):
        #打卡失败则发送打卡失败图片(2.jpg)
        msg.attach(MIMEText('<html><body><h1>ERRO</h1>' +
            '<p><img src="cid:2"></p>' +
            '</body></html>', 'html', 'utf-8'))
        #打开本地文件中的某个图片，并写入将要传入的对象中
        with open(r'2.jpg', 'rb') as f:
            #确认添加的类型为图片格式为jpg，添加头部文件和ID，导入内容，进行编码，链接到邮件
            mine=MIMEBase('image','jpg')
            mine.add_header('Content-ID','2')
            mine.set_payload(f.read())
            encoders.encode_base64(mine)
            msg.attach(mine)

    #确定发送邮箱，授权码，目标邮箱，服务器端口及地址，还有邮箱格式的添加（发送邮箱，目标邮箱，主题）
    from_addr='xxxxxxxxxx@qq.com'#发送邮箱
    password='xxxxxxxxxxxxxxxx'#SMTP授权码
    to_addr='xxxxxxxxxx@qq.com'#接收邮箱
    smtp_server='smtp.qq.com'#此处是QQ邮箱示例
    msg['From']=_format_addr_('发件人昵称<%s>'%from_addr)#自行修改
    msg['To']=_format_addr_('收件人昵称<%s>'%to_addr)#自行修改
    msg['Subject']=Header('主题','utf-8').encode()#自行修改
    #连接向目标端口和服务器，建立安全连接，打出运行日志，登录，发送邮件，退出
    import smtplib
    server=smtplib.SMTP(smtp_server,25)
    server.starttls()
    server.set_debuglevel(1)
    server.login(from_addr,password)
    server.sendmail(from_addr,[to_addr],msg.as_string())
    server.quit
